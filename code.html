The Code for FizzBuzz.

                        
// First get the fizz and buzz values
function getValues() {

let fizzValue = document.getElementById("fizzValue").value;
let buzzValue = document.getElementById("buzzValue").value;

fizzValue = parseInt(fizzValue);
buzzValue = parseInt(buzzValue);

if (Number.isInteger(fizzValue) && Number.isInteger(buzzValue)) 
{let fbvalues = fizzBuzzA(fizzValue, buzzValue);
displayValues(fbvalues);}
 else {alert("You must enter integers!")}

}

// business logic layer - Model
function fizzBuzzA(fizz, buzz) {
let fbvalues = [];

for (let i = 1; i < 101; i++) { if (i % fizz==0 && i % buzz==0) { fbvalues.push("FizzBuzz"); }
 else if (i % fizz==0) {fbvalues.push("fizz"); } 
 else if (i % buzz==0) { fbvalues.push("buzz"); } 
 else { fbvalues.push(i); } } return fbvalues; } 

function displayValues(fbvalues) {
let element = document.getElementById('results');
element.innerHTML = "";

for (let index = 0; index < fbvalues.length; index++)
 { let item=document.createElement("div");
    item.classList.add("col"); item.innerHTML=fbvalues[index];
 if (fbvalues[index]=='fizz' ) {item.classList.add("fizz"); } 
 else if (fbvalues[index]=='buzz' ) { item.classList.add('buzz'); } 
 else if (fbvalues[index]=='FizzBuzz' ) { item.classList.add('fizzbuzz'); } element.appendChild(item); }

This code is a little more complex than The Final Countdown. Using 2 variables, we had to find the divisible options using code and then display the results with styling. This was accomplished using three layers of logic. A function to get the values the user inputed, then the business logic layer, and then the function displaying the results.

Get the Values

This funcion, getValues, gets the numbers inputed by the user, and turns it into a form the next function can work with.

Business Layer Logic

This code is the first time I separated out code into a function all on it's own to do the work, and then call that function elsewhere to use the values. In this layer, the code takes the values from the funciton above, and does the math to determine which numbers get highlighted.
